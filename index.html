API Endpoints Documentation

POST /create-order

Creates a new payment order and returns order_id

Request Body

{ "amount": 50000, "currency": "INR", "receipt": "receipt#1" }

Response

{ "success": true, "order_id": "order_EKwxwAgItmmXdp", "amount": 50000, "currency": "INR" }

Flask Implementation

from flask import Flask, request, jsonify import razorpay app = Flask(__name__) client = razorpay.Client(auth=("rzp_test_key", "rzp_test_secret")) @app.route('/create-order', methods=['POST']) def create_order(): data = { 'amount': request.json['amount'], 'currency': 'INR', 'receipt': f'receipt#{request.json["receipt"]}' } order = client.order.create(data=data) return jsonify({ 'success': True, 'order_id': order['id'], 'amount': order['amount'], 'currency': order['currency'] })
POST /verify-payment

Verifies payment signature for security

Request Body

{ "payment_id": "pay_EKwxwBgDtmmXdq", "order_id": "order_EKwxwAgItmmXdp", "signature": "valid_signature_hash" }

Response

{ "success": true, "message": "Payment verified successfully", "status": "captured" }

Flask Implementation

import hmac import hashlib @app.route('/verify-payment', methods=['POST']) def verify_payment(): payment_id = request.json['payment_id'] order_id = request.json['order_id'] signature = request.json['signature'] # Create signature message = order_id + '|' + payment_id secret = 'your_webhook_secret' generated_signature = hmac.new( secret.encode(), message.encode(), hashlib.sha256 ).hexdigest() if generated_signature == signature: # Save to database save_transaction(payment_id, order_id, 'success') return jsonify({'success': True, 'message': 'Payment verified'}) else: return jsonify({'success': False, 'message': 'Invalid signature'})

Database Schema & Operations

Transactions Table Schema

Column Type Description
id INTEGER PRIMARY KEY Unique transaction ID
order_id VARCHAR(255) Razorpay order ID
payment_id VARCHAR(255) Razorpay payment ID
amount INTEGER Amount in smallest currency unit
currency VARCHAR(10) Currency code (INR, USD, etc.)
status VARCHAR(50) Payment status (pending, success, failed)
payment_method VARCHAR(100) Payment method used
created_at TIMESTAMP Transaction creation time
updated_at TIMESTAMP Last update time

Database Operations

import sqlite3 from datetime import datetime def save_transaction(payment_id, order_id, status, amount=None): conn = sqlite3.connect('payments.db') cursor = conn.cursor() cursor.execute(''' INSERT INTO transactions (payment_id, order_id, amount, currency, status, created_at) VALUES (?, ?, ?, ?, ?, ?) ''', (payment_id, order_id, amount, 'INR', status, datetime.now())) conn.commit() conn.close() return cursor.lastrowid # Create table def create_transactions_table(): conn = sqlite3.connect('payments.db') cursor = conn.cursor() cursor.execute(''' CREATE TABLE IF NOT EXISTS transactions ( id INTEGER PRIMARY KEY AUTOINCREMENT, order_id VARCHAR(255) NOT NULL, payment_id VARCHAR(255), amount INTEGER NOT NULL, currency VARCHAR(10) DEFAULT 'INR', status VARCHAR(50) NOT NULL, payment_method VARCHAR(100), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ) ''') conn.commit() conn.close()

Security Implementation Example

# Environment variables for API keys import os RAZORPAY_KEY_ID = os.environ.get('RAZORPAY_KEY_ID') RAZORPAY_KEY_SECRET = os.environ.get('RAZORPAY_KEY_SECRET') # Webhook signature verification def verify_webhook_signature(payload, signature): expected_signature = hmac.new( RAZORPAY_KEY_SECRET.encode('utf-8'), payload, hashlib.sha256 ).hexdigest() return hmac.compare_digest(expected_signature, signature) # Rate limiting for API endpoints from flask_limiter import Limiter limiter = Limiter(app, key_func=get_remote_address) @app.route('/webhook', methods=['POST']) @limiter.limit("100 per minute") def webhook_handler(): payload = request.get_data() signature = request.headers.get('X-Razorpay-Signature') if not verify_webhook_signature(payload, signature): return jsonify({'error': 'Invalid signature'}), 400 # Process webhook safely return jsonify({'status': 'success'})
